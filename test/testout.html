JFN Examples


For docs see here

Example: simple
integers are nice
 data:
{
  "foo": "bar",
  "bar": 3
}
 schema:
{
  "type": "object",
  "properties": {
    "foo": {
      "type": "string",
      "enum": [
        "foo",
        "bar"
      ],
      "title": "Your title"
    },
    "bar": {
      "type": "integer",
      "description": "integers are nice"
    }
  }
}
Example: type hinting
data:
{
  "foo": "bar",
  "bar": 3
}
 schema:
{
  "type": "object",
  "properties": {
    "foo": {
      "type": "string",
      "typehint": "rich"
    }
  }
}
Example: arrays
data:
{
  "persons": [
    {
      "name": "john doe",
      "age": 12
    }
  ]
}
 schema:
{
  "type": "object",
  "properties": {
    "persons": {
      "type": "array",
      "items": [
        {
          "name": {
            "type": "string"
          },
          "age": {
            "type": "integer"
          }
        }
      ]
    }
  }
}
